\begin{enumerate}
\item If two positive integers $p$ and $q$ can be expressed as $p = 18a^2 b^4$ and $q = 20a^3 b^2$ where $a$ and $b$ are prime numbers, then $\text{LCM}(p, q)$ is:
	\begin{enumerate}    
\item $2a^2 b^2$
    \item $180a^2 b^2$
    \item $12a^2  b^2$
    \item $180a^3 b^4$
	\end{enumerate}
\end{enumerate}
